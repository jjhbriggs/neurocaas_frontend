23class Config(object): 

24 """Advanced configuration for Botocore clients. 

25 

26 :type region_name: str 

27 :param region_name: The region to use in instantiating the client 

28 

29 :type signature_version: str 

30 :param signature_version: The signature version when signing requests. 

31 

32 :type user_agent: str 

33 :param user_agent: The value to use in the User-Agent header. 

34 

35 :type user_agent_extra: str 

36 :param user_agent_extra: The value to append to the current User-Agent 

37 header value. 

38 

39 :type connect_timeout: float or int 

40 :param connect_timeout: The time in seconds till a timeout exception is 

41 thrown when attempting to make a connection. The default is 60 

42 seconds. 

43 

44 :type read_timeout: float or int 

45 :param read_timeout: The time in seconds till a timeout exception is 

46 thrown when attempting to read from a connection. The default is 

47 60 seconds. 

48 

49 :type parameter_validation: bool 

50 :param parameter_validation: Whether parameter validation should occur 

51 when serializing requests. The default is True. You can disable 

52 parameter validation for performance reasons. Otherwise, it's 

53 recommended to leave parameter validation enabled. 

54 

55 :type max_pool_connections: int 

56 :param max_pool_connections: The maximum number of connections to 

57 keep in a connection pool. If this value is not set, the default 

58 value of 10 is used. 

59 

60 :type proxies: dict 

61 :param proxies: A dictionary of proxy servers to use by protocol or 

62 endpoint, e.g.: 

63 {'http': 'foo.bar:3128', 'http://hostname': 'foo.bar:4012'}. 

64 The proxies are used on each request. 

65 

66 :type s3: dict 

67 :param s3: A dictionary of s3 specific configurations. 

68 Valid keys are: 

69 

70 * 'use_accelerate_endpoint' -- Refers to whether to use the S3 

71 Accelerate endpoint. The value must be a boolean. If True, the 

72 client will use the S3 Accelerate endpoint. If the S3 Accelerate 

73 endpoint is being used then the addressing style will always 

74 be virtual. 

75 

76 * 'payload_signing_enabled' -- Refers to whether or not to SHA256 

77 sign sigv4 payloads. By default, this is disabled for streaming 

78 uploads (UploadPart and PutObject). 

79 

80 * 'addressing_style' -- Refers to the style in which to address 

81 s3 endpoints. Values must be a string that equals: 

82 

83 * auto -- Addressing style is chosen for user. Depending 

84 on the configuration of client, the endpoint may be addressed in 

85 the virtual or the path style. Note that this is the default 

86 behavior if no style is specified. 

87 

88 * virtual -- Addressing style is always virtual. The name of the 

89 bucket must be DNS compatible or an exception will be thrown. 

90 Endpoints will be addressed as such: mybucket.s3.amazonaws.com 

91 

92 * path -- Addressing style is always by path. Endpoints will be 

93 addressed as such: s3.amazonaws.com/mybucket 

94 

95 * 'us_east_1_regional_endpoint' - Refers to what S3 endpoint to use 

96 when the region is configured to be us-east-1. Values must be a 

97 string that equals: 

98 

99 * regional -- Use the us-east-1.amazonaws.com endpoint if the 

100 client is configured to use the us-east-1 region. 

101 

102 * legacy -- Use the s3.amazonaws.com endpoint if the client is 

103 configured to use the us-east-1 region. This is the default if 

104 the configuration option is not specified. 

105 

106 

107 :type retries: dict 

108 :param retries: A dictionary for retry specific configurations. 

109 Valid keys are: 

110 

111 * 'total_max_attempts' -- An integer representing the maximum number of 

112 total attempts that will be made on a single request. This includes 

113 the initial request, so a value of 1 indicates that no requests 

114 will be retried. If ``total_max_attempts`` and ``max_attempts`` 

115 are both provided, ``total_max_attempts`` takes precedence. 

116 ``total_max_attempts`` is preferred over ``max_attempts`` because 

117 it maps to the ``AWS_MAX_ATTEMPTS`` environment variable and 

118 the ``max_attempts`` config file value. 

119 * 'max_attempts' -- An integer representing the maximum number of 

120 retry attempts that will be made on a single request. For 

121 example, setting this value to 2 will result in the request 

122 being retried at most two times after the initial request. Setting 

123 this value to 0 will result in no retries ever being attempted on 

124 the initial request. If not provided, the number of retries will 

125 default to whatever is modeled, which is typically four retries. 

126 * 'mode' -- A string representing the type of retry mode botocore 

127 should use. Valid values are: 

128 * ``legacy`` - The pre-existing retry behavior. 

129 * ``standard`` - The standardized set of retry rules. This 

130 will also default to 3 max attempts unless overridden. 

131 * ``adaptive`` - Retries with additional client side throttling. 

132 

133 :type client_cert: str, (str, str) 

134 :param client_cert: The path to a certificate for TLS client authentication. 

135 

136 When a str is provided it is treated as a path to a client certificate 

137 to be used when creating a TLS connection. 

138 

139 If a client key is to be provided alongside the client certificate the 

140 client_cert should be set to a tuple of length two where the first 

141 element is the path to the client certificate and the second element is 

142 the path to the certificate key. 

143 

144 :type inject_host_prefix: bool 

145 :param inject_host_prefix: Whether host prefix injection should occur. 

146 

147 Defaults to True. 

148 

149 Setting this to False disables the injection of operation parameters 

150 into the prefix of the hostname. This is useful for clients providing 

151 custom endpoints that should not have their host prefix modified. 

152 """ 

153 OPTION_DEFAULTS = OrderedDict([ 

154 ('region_name', None), 

155 ('signature_version', None), 

156 ('user_agent', None), 

157 ('user_agent_extra', None), 

158 ('connect_timeout', DEFAULT_TIMEOUT), 

159 ('read_timeout', DEFAULT_TIMEOUT), 

160 ('parameter_validation', True), 

161 ('max_pool_connections', MAX_POOL_CONNECTIONS), 

162 ('proxies', None), 

163 ('s3', None), 

164 ('retries', None), 

165 ('client_cert', None), 

166 ('inject_host_prefix', True), 

167 ('endpoint_discovery_enabled', None), 

168 ])

170 def __init__(self, *args, **kwargs): 

171 self._user_provided_options = self._record_user_provided_options( 

172 args, kwargs) 

173 

174 # Merge the user_provided options onto the default options 

175 config_vars = copy.copy(self.OPTION_DEFAULTS) 

176 config_vars.update(self._user_provided_options) 

177 

178 # Set the attributes based on the config_vars 

179 for key, value in config_vars.items(): 

180 setattr(self, key, value) 

181 

182 # Validate the s3 options 

183 self._validate_s3_configuration(self.s3) 

184 

185 self._validate_retry_configuration(self.retries)

187 def _record_user_provided_options(self, args, kwargs): 

188 option_order = list(self.OPTION_DEFAULTS) 

189 user_provided_options = {} 

190 

191 # Iterate through the kwargs passed through to the constructor and 

192 # map valid keys to the dictionary 

193 for key, value in kwargs.items(): 

194 if key in self.OPTION_DEFAULTS: 

195 user_provided_options[key] = value 

196 # The key must exist in the available options 

197 else: 

198 raise TypeError( 

199 'Got unexpected keyword argument \'%s\'' % key) 

200 

201 # The number of args should not be longer than the allowed 

202 # options 

203 if len(args) > len(option_order): 

204 raise TypeError( 

205 'Takes at most %s arguments (%s given)' % ( 

206 len(option_order), len(args))) 

207 

208 # Iterate through the args passed through to the constructor and map 

209 # them to appropriate keys. 

210 for i, arg in enumerate(args): 

211 # If it a kwarg was specified for the arg, then error out 

212 if option_order[i] in user_provided_options: 

213 raise TypeError( 

214 'Got multiple values for keyword argument \'%s\'' % ( 

215 option_order[i])) 

216 user_provided_options[option_order[i]] = arg 

217 

218 return user_provided_options

220 def _validate_s3_configuration(self, s3): 

221 if s3 is not None: 

222 addressing_style = s3.get('addressing_style') 

223 if addressing_style not in ['virtual', 'auto', 'path', None]: 

224 raise InvalidS3AddressingStyleError( 

225 s3_addressing_style=addressing_style)

227 def _validate_retry_configuration(self, retries): 

228 if retries is not None: 

229 for key, value in retries.items(): 

230 if key not in ['max_attempts', 'mode', 'total_max_attempts']: 

231 raise InvalidRetryConfigurationError( 

232 retry_config_option=key) 

233 if key == 'max_attempts' and value < 0: 

234 raise InvalidMaxRetryAttemptsError( 

235 provided_max_attempts=value, 

236 min_value=0, 

237 ) 

238 if key == 'total_max_attempts' and value < 1: 

239 raise InvalidMaxRetryAttemptsError( 

240 provided_max_attempts=value, 

241 min_value=1, 

242 ) 

243 if key == 'mode' and value not in ['legacy', 'standard', 

244 'adaptive']: 

245 raise InvalidRetryModeError( 

246 provided_retry_mode=value 

247 )

249 def merge(self, other_config): 

250 """Merges the config object with another config object 

251 

252 This will merge in all non-default values from the provided config 

253 and return a new config object 

254 

255 :type other_config: botocore.config.Config 

256 :param other config: Another config object to merge with. The values 

257 in the provided config object will take precedence in the merging 

258 

259 :returns: A config object built from the merged values of both 

260 config objects. 

261 """ 

262 # Make a copy of the current attributes in the config object. 

263 config_options = copy.copy(self._user_provided_options) 

264 

265 # Merge in the user provided options from the other config 

266 config_options.update(other_config._user_provided_options) 

267 

268 # Return a new config object with the merged properties. 

269 return Config(**config_options)
