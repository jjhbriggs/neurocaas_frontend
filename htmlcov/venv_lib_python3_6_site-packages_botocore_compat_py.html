111 def unquote_str(value, encoding='utf-8'): 

112 # In python2, unquote() gives us a string back that has the urldecoded 

113 # bits, but not the unicode parts. We need to decode this manually. 

114 # unquote has special logic in which if it receives a unicode object it 

115 # will decode it to latin1. This is hard coded. To avoid this, we'll 

116 # encode the string with the passed in encoding before trying to 

117 # unquote it. 

118 byte_string = value.encode(encoding) 

119 return unquote_plus(byte_string).decode(encoding)

238def _windows_shell_split(s): 

239 """Splits up a windows command as the built-in command parser would. 

240 

241 Windows has potentially bizarre rules depending on where you look. When 

242 spawning a process via the Windows C runtime (which is what python does 

243 when you call popen) the rules are as follows: 

244 

245 https://docs.microsoft.com/en-us/cpp/cpp/parsing-cpp-command-line-arguments 

246 

247 To summarize: 

248 

249 * Only space and tab are valid delimiters 

250 * Double quotes are the only valid quotes 

251 * Backslash is interpreted literally unless it is part of a chain that 

252 leads up to a double quote. Then the backslashes escape the backslashes, 

253 and if there is an odd number the final backslash escapes the quote. 

254 

255 :param s: The command string to split up into parts. 

256 :return: A list of command components. 

257 """ 

258 if not s: 

259 return [] 

260 

261 components = [] 

262 buff = [] 

263 is_quoted = False 

264 num_backslashes = 0 

265 for character in s: 

266 if character == '\\': 

267 # We can't simply append backslashes because we don't know if 

268 # they are being used as escape characters or not. Instead we 

269 # keep track of how many we've encountered and handle them when 

270 # we encounter a different character. 

271 num_backslashes += 1 

272 elif character == '"': 

273 if num_backslashes > 0: 

274 # The backslashes are in a chain leading up to a double 

275 # quote, so they are escaping each other. 

276 buff.append('\\' * int(floor(num_backslashes / 2))) 

277 remainder = num_backslashes % 2 

278 num_backslashes = 0 

279 if remainder == 1: 

280 # The number of backslashes is uneven, so they are also 

281 # escaping the double quote, so it needs to be added to 

282 # the current component buffer. 

283 buff.append('"') 

284 continue 

285 

286 # We've encountered a double quote that is not escaped, 

287 # so we toggle is_quoted. 

288 is_quoted = not is_quoted 

289 

290 # If there are quotes, then we may want an empty string. To be 

291 # safe, we add an empty string to the buffer so that we make 

292 # sure it sticks around if there's nothing else between quotes. 

293 # If there is other stuff between quotes, the empty string will 

294 # disappear during the joining process. 

295 buff.append('') 

296 elif character in [' ', '\t'] and not is_quoted: 

297 # Since the backslashes aren't leading up to a quote, we put in 

298 # the exact number of backslashes. 

299 if num_backslashes > 0: 

300 buff.append('\\' * num_backslashes) 

301 num_backslashes = 0 

302 

303 # Excess whitespace is ignored, so only add the components list 

304 # if there is anything in the buffer. 

305 if buff: 

306 components.append(''.join(buff)) 

307 buff = [] 

308 else: 

309 # Since the backslashes aren't leading up to a quote, we put in 

310 # the exact number of backslashes. 

311 if num_backslashes > 0: 

312 buff.append('\\' * num_backslashes) 

313 num_backslashes = 0 

314 buff.append(character) 

315 

316 # Quotes must be terminated. 

317 if is_quoted: 

318 raise ValueError('No closing quotation in string: %s' % s) 

319 

320 # There may be some leftover backslashes, so we need to add them in. 

321 # There's no quote so we add the exact number. 

322 if num_backslashes > 0: 

323 buff.append('\\' * num_backslashes) 

324 

325 # Add the final component in if there is anything in the buffer. 

326 if buff: 

327 components.append(''.join(buff)) 

328 

329 return components
