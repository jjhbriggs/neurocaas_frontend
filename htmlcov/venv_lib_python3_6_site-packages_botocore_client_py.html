224 def _set_s3_presign_signature_version(self, client_meta, 

225 client_config, scoped_config): 

226 # This will return the manually configured signature version, or None 

227 # if none was manually set. If a customer manually sets the signature 

228 # version, we always want to use what they set. 

229 provided_signature_version = _get_configured_signature_version( 

230 's3', client_config, scoped_config) 

231 if provided_signature_version is not None: 

232 return 

233 

234 # Check to see if the region is a region that we know about. If we 

235 # don't know about a region, then we can safely assume it's a new 

236 # region that is sigv4 only, since all new S3 regions only allow sigv4. 

237 # The only exception is aws-global. This is a pseudo-region for the 

238 # global endpoint, we should respect the signature versions it 

239 # supports, which includes v2. 

240 regions = self._endpoint_resolver.get_available_endpoints( 

241 's3', client_meta.partition) 

242 if client_meta.region_name != 'aws-global' and \ 

243 client_meta.region_name not in regions: 

244 return 

245 

246 # If it is a region we know about, we want to default to sigv2, so here 

247 # we check to see if it is available. 

248 endpoint = self._endpoint_resolver.construct_endpoint( 

249 's3', client_meta.region_name) 

250 signature_versions = endpoint['signatureVersions'] 

251 if 's3' not in signature_versions: 

252 return 

253 

254 # We now know that we're in a known region that supports sigv2 and 

255 # the customer hasn't set a signature version so we default the 

256 # signature version to sigv2. 

257 client_meta.events.register( 

258 'choose-signer.s3', self._default_s3_presign_to_sigv2)

334class ClientEndpointBridge(object): 

335 """Bridges endpoint data and client creation 

336 

337 This class handles taking out the relevant arguments from the endpoint 

338 resolver and determining which values to use, taking into account any 

339 client configuration options and scope configuration options. 

340 

341 This class also handles determining what, if any, region to use if no 

342 explicit region setting is provided. For example, Amazon S3 client will 

343 utilize "us-east-1" by default if no region can be resolved.""" 

344 

345 DEFAULT_ENDPOINT = '{service}.{region}.amazonaws.com' 

346 _DUALSTACK_ENABLED_SERVICES = ['s3', 's3-control'] 

347 

348 def __init__(self, endpoint_resolver, scoped_config=None, 

349 client_config=None, default_endpoint=None, 

350 service_signing_name=None): 

351 self.service_signing_name = service_signing_name 

352 self.endpoint_resolver = endpoint_resolver 

353 self.scoped_config = scoped_config 

354 self.client_config = client_config 

355 self.default_endpoint = default_endpoint or self.DEFAULT_ENDPOINT

357 def resolve(self, service_name, region_name=None, endpoint_url=None, 

358 is_secure=True): 

359 region_name = self._check_default_region(service_name, region_name) 

360 resolved = self.endpoint_resolver.construct_endpoint( 

361 service_name, region_name) 

362 if resolved: 

363 return self._create_endpoint( 

364 resolved, service_name, region_name, endpoint_url, is_secure) 

365 else: 

366 return self._assume_endpoint(service_name, region_name, 

367 endpoint_url, is_secure)

369 def _check_default_region(self, service_name, region_name): 

370 if region_name is not None: 

371 return region_name 

372 # Use the client_config region if no explicit region was provided. 

373 if self.client_config and self.client_config.region_name is not None: 

374 return self.client_config.region_name

376 def _create_endpoint(self, resolved, service_name, region_name, 

377 endpoint_url, is_secure): 

378 explicit_region = region_name is not None 

379 region_name, signing_region = self._pick_region_values( 

380 resolved, region_name, endpoint_url) 

381 if endpoint_url is None: 

382 if self._is_s3_dualstack_mode(service_name): 

383 endpoint_url = self._create_dualstack_endpoint( 

384 service_name, region_name, 

385 resolved['dnsSuffix'], is_secure, explicit_region) 

386 else: 

387 # Use the sslCommonName over the hostname for Python 2.6 compat. 

388 hostname = resolved.get('sslCommonName', resolved.get('hostname')) 

389 endpoint_url = self._make_url(hostname, is_secure, 

390 resolved.get('protocols', [])) 

391 signature_version = self._resolve_signature_version( 

392 service_name, resolved) 

393 signing_name = self._resolve_signing_name(service_name, resolved) 

394 return self._create_result( 

395 service_name=service_name, region_name=region_name, 

396 signing_region=signing_region, signing_name=signing_name, 

397 endpoint_url=endpoint_url, metadata=resolved, 

398 signature_version=signature_version)

400 def _is_s3_dualstack_mode(self, service_name): 

401 if service_name not in self._DUALSTACK_ENABLED_SERVICES: 

402 return False 

403 # TODO: This normalization logic is duplicated from the 

404 # ClientArgsCreator class. Consolidate everything to 

405 # ClientArgsCreator. _resolve_signature_version also has similarly 

406 # duplicated logic. 

407 client_config = self.client_config 

408 if client_config is not None and client_config.s3 is not None and \ 

409 'use_dualstack_endpoint' in client_config.s3: 

410 # Client config trumps scoped config. 

411 return client_config.s3['use_dualstack_endpoint'] 

412 if self.scoped_config is None: 

413 return False 

414 enabled = self.scoped_config.get('s3', {}).get( 

415 'use_dualstack_endpoint', False) 

416 if enabled in [True, 'True', 'true']: 

417 return True 

418 return False

420 def _create_dualstack_endpoint(self, service_name, region_name, 

421 dns_suffix, is_secure, explicit_region): 

422 if not explicit_region and region_name == 'aws-global': 

423 # If the region_name passed was not explicitly set, default to 

424 # us-east-1 instead of the modeled default aws-global. Dualstack 

425 # does not support aws-global 

426 region_name = 'us-east-1' 

427 hostname = '{service}.dualstack.{region}.{dns_suffix}'.format( 

428 service=service_name, region=region_name, 

429 dns_suffix=dns_suffix) 

430 # Dualstack supports http and https so were hardcoding this value for 

431 # now. This can potentially move into the endpoints.json file. 

432 return self._make_url(hostname, is_secure, ['http', 'https'])

434 def _assume_endpoint(self, service_name, region_name, endpoint_url, 

435 is_secure): 

436 if endpoint_url is None: 

437 # Expand the default hostname URI template. 

438 hostname = self.default_endpoint.format( 

439 service=service_name, region=region_name) 

440 endpoint_url = self._make_url(hostname, is_secure, 

441 ['http', 'https']) 

442 logger.debug('Assuming an endpoint for %s, %s: %s', 

443 service_name, region_name, endpoint_url) 

444 # We still want to allow the user to provide an explicit version. 

445 signature_version = self._resolve_signature_version( 

446 service_name, {'signatureVersions': ['v4']}) 

447 signing_name = self._resolve_signing_name(service_name, resolved={}) 

448 return self._create_result( 

449 service_name=service_name, region_name=region_name, 

450 signing_region=region_name, signing_name=signing_name, 

451 signature_version=signature_version, endpoint_url=endpoint_url, 

452 metadata={})

467 def _make_url(self, hostname, is_secure, supported_protocols): 

468 if is_secure and 'https' in supported_protocols: 

469 scheme = 'https' 

470 else: 

471 scheme = 'http' 

472 return '%s://%s' % (scheme, hostname)

474 def _resolve_signing_name(self, service_name, resolved): 

475 # CredentialScope overrides everything else. 

476 if 'credentialScope' in resolved \ 

477 and 'service' in resolved['credentialScope']: 

478 return resolved['credentialScope']['service'] 

479 # Use the signingName from the model if present. 

480 if self.service_signing_name: 

481 return self.service_signing_name 

482 # Just assume is the same as the service name. 

483 return service_name

485 def _pick_region_values(self, resolved, region_name, endpoint_url): 

486 signing_region = region_name 

487 if endpoint_url is None: 

488 # Do not use the region name or signing name from the resolved 

489 # endpoint if the user explicitly provides an endpoint_url. This 

490 # would happen if we resolve to an endpoint where the service has 

491 # a "defaults" section that overrides all endpoint with a single 

492 # hostname and credentialScope. This has been the case historically 

493 # for how STS has worked. The only way to resolve an STS endpoint 

494 # was to provide a region_name and an endpoint_url. In that case, 

495 # we would still resolve an endpoint, but we would not use the 

496 # resolved endpointName or signingRegion because we want to allow 

497 # custom endpoints. 

498 region_name = resolved['endpointName'] 

499 signing_region = region_name 

500 if 'credentialScope' in resolved \ 

501 and 'region' in resolved['credentialScope']: 

502 signing_region = resolved['credentialScope']['region'] 

503 return region_name, signing_region

505 def _resolve_signature_version(self, service_name, resolved): 

506 configured_version = _get_configured_signature_version( 

507 service_name, self.client_config, self.scoped_config) 

508 if configured_version is not None: 

509 return configured_version 

510 

511 # Pick a signature version from the endpoint metadata if present. 

512 if 'signatureVersions' in resolved: 

513 potential_versions = resolved['signatureVersions'] 

514 if service_name == 's3': 

515 return 's3v4' 

516 if 'v4' in potential_versions: 

517 return 'v4' 

518 # Now just iterate over the signature versions in order until we 

519 # find the first one that is known to Botocore. 

520 for known in potential_versions: 

521 if known in AUTH_TYPE_MAPS: 

522 return known 

523 raise UnknownSignatureVersionError( 

524 signature_version=resolved.get('signatureVersions'))

581 def _make_api_call(self, operation_name, api_params): 

582 operation_model = self._service_model.operation_model(operation_name) 

583 service_name = self._service_model.service_name 

584 history_recorder.record('API_CALL', { 

585 'service': service_name, 

586 'operation': operation_name, 

587 'params': api_params, 

588 }) 

589 if operation_model.deprecated: 

590 logger.debug('Warning: %s.%s() is deprecated', 

591 service_name, operation_name) 

592 request_context = { 

593 'client_region': self.meta.region_name, 

594 'client_config': self.meta.config, 

595 'has_streaming_input': operation_model.has_streaming_input, 

596 'auth_type': operation_model.auth_type, 

597 } 

598 request_dict = self._convert_to_request_dict( 

599 api_params, operation_model, context=request_context) 

600 

601 service_id = self._service_model.service_id.hyphenize() 

602 handler, event_response = self.meta.events.emit_until_response( 

603 'before-call.{service_id}.{operation_name}'.format( 

604 service_id=service_id, 

605 operation_name=operation_name), 

606 model=operation_model, params=request_dict, 

607 request_signer=self._request_signer, context=request_context) 

608 

609 if event_response is not None: 

610 http, parsed_response = event_response 

611 else: 

612 http, parsed_response = self._make_request( 

613 operation_model, request_dict, request_context) 

614 

615 self.meta.events.emit( 

616 'after-call.{service_id}.{operation_name}'.format( 

617 service_id=service_id, 

618 operation_name=operation_name), 

619 http_response=http, parsed=parsed_response, 

620 model=operation_model, context=request_context 

621 ) 

622 

623 if http.status_code >= 300: 

624 error_code = parsed_response.get("Error", {}).get("Code") 

625 error_class = self.exceptions.from_code(error_code) 

626 raise error_class(parsed_response, operation_name) 

627 else: 

628 return parsed_response

630 def _make_request(self, operation_model, request_dict, request_context): 

631 try: 

632 return self._endpoint.make_request(operation_model, request_dict) 

633 except Exception as e: 

634 self.meta.events.emit( 

635 'after-call-error.{service_id}.{operation_name}'.format( 

636 service_id=self._service_model.service_id.hyphenize(), 

637 operation_name=operation_model.name), 

638 exception=e, context=request_context 

639 ) 

640 raise

642 def _convert_to_request_dict(self, api_params, operation_model, 

643 context=None): 

644 api_params = self._emit_api_params( 

645 api_params, operation_model, context) 

646 request_dict = self._serializer.serialize_to_request( 

647 api_params, operation_model) 

648 if not self._client_config.inject_host_prefix: 

649 request_dict.pop('host_prefix', None) 

650 prepare_request_dict(request_dict, endpoint_url=self._endpoint.host, 

651 user_agent=self._client_config.user_agent,

655 def _emit_api_params(self, api_params, operation_model, context): 

656 # Given the API params provided by the user and the operation_model 

657 # we can serialize the request to a request_dict. 

658 operation_name = operation_model.name 

659 

660 # Emit an event that allows users to modify the parameters at the 

661 # beginning of the method. It allows handlers to modify existing 

662 # parameters or return a new set of parameters to use. 

663 service_id = self._service_model.service_id.hyphenize() 

664 responses = self.meta.events.emit( 

665 'provide-client-params.{service_id}.{operation_name}'.format( 

666 service_id=service_id, 

667 operation_name=operation_name), 

668 params=api_params, model=operation_model, context=context) 

669 api_params = first_non_none_response(responses, default=api_params) 

670 

671 event_name = ( 

672 'before-parameter-build.{service_id}.{operation_name}') 

673 self.meta.events.emit( 

674 event_name.format( 

675 service_id=service_id, 

676 operation_name=operation_name), 

677 params=api_params, model=operation_model, context=context) 

678 return api_params

680 def get_paginator(self, operation_name): 

681 """Create a paginator for an operation. 

682 

683 :type operation_name: string 

684 :param operation_name: The operation name. This is the same name 

685 as the method name on the client. For example, if the 

686 method name is ``create_foo``, and you'd normally invoke the 

687 operation as ``client.create_foo(**kwargs)``, if the 

688 ``create_foo`` operation can be paginated, you can use the 

689 call ``client.get_paginator("create_foo")``. 

690 

691 :raise OperationNotPageableError: Raised if the operation is not 

692 pageable. You can use the ``client.can_paginate`` method to 

693 check if an operation is pageable. 

694 

695 :rtype: L{botocore.paginate.Paginator} 

696 :return: A paginator object. 

697 

698 """ 

699 if not self.can_paginate(operation_name): 

700 raise OperationNotPageableError(operation_name=operation_name) 

701 else: 

702 actual_operation_name = self._PY_TO_OP_NAME[operation_name] 

703 

704 # Create a new paginate method that will serve as a proxy to 

705 # the underlying Paginator.paginate method. This is needed to 

706 # attach a docstring to the method. 

707 def paginate(self, **kwargs): 

708 return Paginator.paginate(self, **kwargs) 

709 

710 paginator_config = self._cache['page_config'][ 

711 actual_operation_name] 

712 # Add the docstring for the paginate method. 

713 paginate.__doc__ = PaginatorDocstring( 

714 paginator_name=actual_operation_name, 

715 event_emitter=self.meta.events, 

716 service_model=self.meta.service_model, 

717 paginator_config=paginator_config, 

718 include_signature=False 

719 ) 

720 

721 # Rename the paginator class based on the type of paginator. 

722 paginator_class_name = str('%s.Paginator.%s' % ( 

723 get_service_module_name(self.meta.service_model), 

724 actual_operation_name)) 

725 

726 # Create the new paginator class 

727 documented_paginator_cls = type( 

728 paginator_class_name, (Paginator,), {'paginate': paginate}) 

729 

730 operation_model = self._service_model.operation_model(actual_operation_name) 

731 paginator = documented_paginator_cls( 

732 getattr(self, operation_name), 

733 paginator_config, 

734 operation_model) 

735 return paginator

737 def can_paginate(self, operation_name): 

738 """Check if an operation can be paginated. 

739 

740 :type operation_name: string 

741 :param operation_name: The operation name. This is the same name 

742 as the method name on the client. For example, if the 

743 method name is ``create_foo``, and you'd normally invoke the 

744 operation as ``client.create_foo(**kwargs)``, if the 

745 ``create_foo`` operation can be paginated, you can use the 

746 call ``client.get_paginator("create_foo")``. 

747 

748 :return: ``True`` if the operation can be paginated, 

749 ``False`` otherwise. 

750 

751 """ 

752 if 'page_config' not in self._cache: 

753 try: 

754 page_config = self._loader.load_service_model( 

755 self._service_model.service_name, 

756 'paginators-1', 

757 self._service_model.api_version)['pagination'] 

758 self._cache['page_config'] = page_config 

759 except DataNotFoundError: 

760 self._cache['page_config'] = {} 

761 actual_operation_name = self._PY_TO_OP_NAME[operation_name] 

762 return actual_operation_name in self._cache['page_config']

777 """Returns an object that can wait for some condition. 

778 

779 :type waiter_name: str 

780 :param waiter_name: The name of the waiter to get. See the waiters 

781 section of the service docs for a list of available waiters.

783 :returns: The specified waiter object.

786 config = self._get_waiter_config()

868def _get_configured_signature_version(service_name, client_config, 

869 scoped_config): 

870 """ 

871 Gets the manually configured signature version. 

872 

873 :returns: the customer configured signature version, or None if no 

874 signature version was configured. 

875 """ 

876 # Client config overrides everything. 

877 if client_config and client_config.signature_version is not None: 

878 return client_config.signature_version 

879 

880 # Scoped config overrides picking from the endpoint metadata. 

881 if scoped_config is not None: 

882 # A given service may have service specific configuration in the 

883 # config file, so we need to check there as well. 

884 service_config = scoped_config.get(service_name) 

885 if service_config is not None and isinstance(service_config, dict): 

886 version = service_config.get('signature_version') 

887 if version: 

888 logger.debug( 

889 "Switching signature version for service %s " 

890 "to version %s based on config file override.", 

891 service_name, version) 

892 return version 

893 return None
