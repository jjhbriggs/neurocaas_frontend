26def validate_parameters(params, shape): 

27 """Validates input parameters against a schema. 

28 

29 This is a convenience function that validates parameters against a schema. 

30 You can also instantiate and use the ParamValidator class directly if you 

31 want more control. 

32 

33 If there are any validation errors then a ParamValidationError 

34 will be raised. If there are no validation errors than no exception 

35 is raised and a value of None is returned. 

36 

37 :param params: The user provided input parameters. 

38 

39 :type shape: botocore.model.Shape 

40 :param shape: The schema which the input parameters should 

41 adhere to. 

42 

43 :raise: ParamValidationError 

44 

45 """ 

46 validator = ParamValidator() 

47 report = validator.validate(params, shape) 

48 if report.has_errors(): 

49 raise ParamValidationError(report=report.generate_report())

70def range_check(name, value, shape, error_type, errors): 

71 failed = False 

72 min_allowed = float('-inf') 

73 max_allowed = float('inf') 

74 if 'min' in shape.metadata: 

75 min_allowed = shape.metadata['min'] 

76 if value < min_allowed: 

77 failed = True 

78 elif hasattr(shape, 'serialization'): 

79 # Members that can be bound to the host have an implicit min of 1 

80 if shape.serialization.get('hostLabel'): 

81 min_allowed = 1 

82 if value < min_allowed: 

83 failed = True 

84 if failed: 

85 errors.report(name, error_type, param=value, 

86 valid_range=[min_allowed, max_allowed])

98 def generate_report(self): 

99 error_messages = [] 

100 for error in self._errors: 

101 error_messages.append(self._format_error(error)) 

102 return '\n'.join(error_messages) 

103 

104 def _format_error(self, error): 

105 error_type, name, additional = error 

106 name = self._get_name(name) 

107 if error_type == 'missing required field': 

108 return 'Missing required parameter in %s: "%s"' % ( 

109 name, additional['required_name']) 

110 elif error_type == 'unknown field': 

111 return 'Unknown parameter in %s: "%s", must be one of: %s' % ( 

112 name, additional['unknown_param'], 

113 ', '.join(additional['valid_names'])) 

114 elif error_type == 'invalid type': 

115 return 'Invalid type for parameter %s, value: %s, type: %s, ' \ 

116 'valid types: %s' % (name, additional['param'], 

117 str(type(additional['param'])), 

118 ', '.join(additional['valid_types'])) 

119 elif error_type == 'invalid range': 

120 min_allowed = additional['valid_range'][0] 

121 max_allowed = additional['valid_range'][1] 

122 return ('Invalid range for parameter %s, value: %s, valid range: ' 

123 '%s-%s' % (name, additional['param'], 

124 min_allowed, max_allowed)) 

125 elif error_type == 'invalid length': 

126 min_allowed = additional['valid_range'][0] 

127 max_allowed = additional['valid_range'][1] 

128 return ('Invalid length for parameter %s, value: %s, valid range: ' 

129 '%s-%s' % (name, additional['param'], 

130 min_allowed, max_allowed)) 

131 elif error_type == 'unable to encode to json': 

132 return 'Invalid parameter %s must be json serializable: %s' \ 

133 % (name, additional['type_error'])

147class ParamValidator(object): 

148 """Validates parameters against a shape model."""

150 def validate(self, params, shape): 

151 """Validate parameters against a shape model. 

152 

153 This method will validate the parameters against a provided shape model. 

154 All errors will be collected before returning to the caller. This means 

155 that this method will not stop at the first error, it will return all 

156 possible errors. 

157 

158 :param params: User provided dict of parameters 

159 :param shape: A shape model describing the expected input. 

160 

161 :return: A list of errors. 

162 

163 """ 

164 errors = ValidationErrors() 

165 self._validate(params, shape, errors, name='') 

166 return errors

168 def _check_special_validation_cases(self, shape): 

169 if is_json_value_header(shape): 

170 return self._validate_jsonvalue_string

172 def _validate(self, params, shape, errors, name): 

173 special_validator = self._check_special_validation_cases(shape) 

174 if special_validator: 

175 special_validator(params, shape, errors, name) 

176 else: 

177 getattr(self, '_validate_%s' % shape.type_name)( 

178 params, shape, errors, name)

180 def _validate_jsonvalue_string(self, params, shape, errors, name): 

181 # Check to see if a value marked as a jsonvalue can be dumped to 

182 # a json string. 

183 try: 

184 json.dumps(params) 

185 except (ValueError, TypeError) as e: 

186 errors.report(name, 'unable to encode to json', type_error=e)

188 @type_check(valid_types=(dict,)) 

189 def _validate_structure(self, params, shape, errors, name): 

190 # Validate required fields. 

191 for required_member in shape.metadata.get('required', []): 

192 if required_member not in params: 

193 errors.report(name, 'missing required field', 

194 required_name=required_member, user_params=params) 

195 members = shape.members 

196 known_params = [] 

197 # Validate known params. 

198 for param in params: 

199 if param not in members: 

200 errors.report(name, 'unknown field', unknown_param=param, 

201 valid_names=list(members)) 

202 else: 

203 known_params.append(param) 

204 # Validate structure members. 

205 for param in known_params: 

206 self._validate(params[param], shape.members[param], 

207 errors, '%s.%s' % (name, param)) 

208 

209 @type_check(valid_types=six.string_types) 

210 def _validate_string(self, param, shape, errors, name): 

211 # Validate range. For a string, the min/max contraints 

212 # are of the string length. 

213 # Looks like: 

214 # "WorkflowId":{ 

215 # "type":"string", 

216 # "min":1, 

217 # "max":256 

218 # } 

219 range_check(name, len(param), shape, 'invalid length', errors)

221 @type_check(valid_types=(list, tuple)) 

222 def _validate_list(self, param, shape, errors, name): 

223 member_shape = shape.member 

224 range_check(name, len(param), shape, 'invalid length', errors) 

225 for i, item in enumerate(param): 

226 self._validate(item, member_shape, errors, '%s[%s]' % (name, i))

228 @type_check(valid_types=(dict,)) 

229 def _validate_map(self, param, shape, errors, name): 

230 key_shape = shape.key 

231 value_shape = shape.value 

232 for key, value in param.items():

267 # We don't use @type_check because datetimes are a bit
