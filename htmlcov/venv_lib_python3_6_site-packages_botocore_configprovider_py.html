25#: A default dictionary that maps the logical names for session variables 

26#: to the specific environment variables and configuration file names 

27#: that contain the values for these variables. 

28#: When creating a new Session object, you can pass in your own dictionary 

29#: to remap the logical names or to add new logical names. You can then 

30#: get the current value for these variables by using the 

31#: ``get_config_variable`` method of the :class:`botocore.session.Session` 

32#: class. 

33#: These form the keys of the dictionary. The values in the dictionary 

34#: are tuples of (<config_name>, <environment variable>, <default value>, 

35#: <conversion func>). 

36#: The conversion func is a function that takes the configuration value 

37#: as an argument and returns the converted value. If this value is 

38#: None, then the configuration value is returned unmodified. This 

39#: conversion function can be used to type convert config values to 

40#: values other than the default values of strings. 

41#: The ``profile`` and ``config_file`` variables should always have a 

42#: None value for the first entry in the tuple because it doesn't make 

43#: sense to look inside the config file for the location of the config 

44#: file or for the default profile to use. 

45#: The ``config_name`` is the name to look for in the configuration file, 

46#: the ``env var`` is the OS environment variable (``os.environ``) to 

47#: use, and ``default_value`` is the value to use if no value is otherwise 

48#: found.

148class ConfigChainFactory(object): 

149 """Factory class to create our most common configuration chain case. 

150 

151 This is a convenience class to construct configuration chains that follow 

152 our most common pattern. This is to prevent ordering them incorrectly, 

153 and to make the config chain construction more readable. 

154 """ 

155 def __init__(self, session, environ=None): 

156 """Initialize a ConfigChainFactory. 

157 

158 :type session: :class:`botocore.session.Session` 

159 :param session: This is the session that should be used to look up 

160 values from the config file. 

161 

162 :type environ: dict 

163 :param environ: A mapping to use for environment variables. If this 

164 is not provided it will default to use os.environ. 

165 """ 

166 self._session = session 

167 if environ is None: 

168 environ = os.environ 

169 self._environ = environ 

170 

171 def create_config_chain(self, instance_name=None, env_var_names=None, 

172 config_property_names=None, default=None, 

173 conversion_func=None): 

174 """Build a config chain following the standard botocore pattern. 

175 

176 In botocore most of our config chains follow the the precendence: 

177 session_instance_variables, environment, config_file, default_value. 

178 

179 This is a convenience function for creating a chain that follow 

180 that precendence. 

181 

182 :type instance_name: str 

183 :param instance_name: This indicates what session instance variable 

184 corresponds to this config value. If it is None it will not be 

185 added to the chain. 

186 

187 :type env_var_names: str or list of str or None 

188 :param env_var_names: One or more environment variable names to 

189 search for this value. They are searched in order. If it is None 

190 it will not be added to the chain. 

191 

192 :type config_property_names: str/tuple or list of str/tuple or None 

193 :param config_property_names: One of more strings or tuples 

194 representing the name of the key in the config file for this 

195 config option. They are searched in order. If it is None it will 

196 not be added to the chain. 

197 

198 :type default: Any 

199 :param default: Any constant value to be returned. 

200 

201 :type conversion_func: None or callable 

202 :param conversion_func: If this value is None then it has no effect on 

203 the return type. Otherwise, it is treated as a function that will 

204 conversion_func our provided type. 

205 

206 :rvalue: ConfigChain 

207 :returns: A ConfigChain that resolves in the order env_var_names -> 

208 config_property_name -> default. Any values that were none are 

209 omitted form the chain. 

210 """ 

211 providers = [] 

212 if instance_name is not None: 

213 providers.append( 

214 InstanceVarProvider( 

215 instance_var=instance_name, 

216 session=self._session 

217 ) 

218 ) 

219 if env_var_names is not None: 

220 providers.extend(self._get_env_providers(env_var_names)) 

221 if config_property_names is not None: 

222 providers.extend( 

223 self._get_scoped_config_providers(config_property_names) 

224 ) 

225 if default is not None: 

226 providers.append(ConstantProvider(value=default)) 

227 

228 return ChainProvider( 

229 providers=providers, 

230 conversion_func=conversion_func, 

231 )

233 def _get_env_providers(self, env_var_names): 

234 env_var_providers = [] 

235 if not isinstance(env_var_names, list): 

236 env_var_names = [env_var_names] 

237 for env_var_name in env_var_names: 

238 env_var_providers.append( 

239 EnvironmentProvider(name=env_var_name, env=self._environ) 

240 ) 

241 return env_var_providers 

242 

243 def _get_scoped_config_providers(self, config_property_names): 

244 scoped_config_providers = [] 

245 if not isinstance(config_property_names, list): 

246 config_property_names = [config_property_names] 

247 for config_property_name in config_property_names: 

248 scoped_config_providers.append( 

249 ScopedConfigProvider( 

250 config_var_name=config_property_name, 

251 session=self._session, 

252 ) 

253 ) 

254 return scoped_config_providers

257class ConfigValueStore(object): 

258 """The ConfigValueStore object stores configuration values.""" 

259 def __init__(self, mapping=None): 

260 """Initialize a ConfigValueStore. 

261 

262 :type mapping: dict 

263 :param mapping: The mapping parameter is a map of string to a subclass 

264 of BaseProvider. When a config variable is asked for via the 

265 get_config_variable method, the corresponding provider will be 

266 invoked to load the value. 

267 """ 

268 self._overrides = {} 

269 self._mapping = {} 

270 if mapping is not None: 

271 for logical_name, provider in mapping.items(): 

272 self.set_config_provider(logical_name, provider)

274 def get_config_variable(self, logical_name): 

275 """ 

276 Retrieve the value associeated with the specified logical_name 

277 from the corresponding provider. If no value is found None will 

278 be returned. 

279 

280 :type logical_name: str 

281 :param logical_name: The logical name of the session variable 

282 you want to retrieve. This name will be mapped to the 

283 appropriate environment variable name for this session as 

284 well as the appropriate config file entry. 

285 

286 :returns: value of variable or None if not defined. 

287 """ 

288 if logical_name in self._overrides: 

289 return self._overrides[logical_name] 

290 if logical_name not in self._mapping: 

291 return None 

292 provider = self._mapping[logical_name] 

293 return provider.provide()

295 def set_config_variable(self, logical_name, value): 

296 """Set a configuration variable to a specific value. 

297 

298 By using this method, you can override the normal lookup 

299 process used in ``get_config_variable`` by explicitly setting 

300 a value. Subsequent calls to ``get_config_variable`` will 

301 use the ``value``. This gives you per-session specific 

302 configuration values. 

303 

304 :: 

305 >>> # Assume logical name 'foo' maps to env var 'FOO' 

306 >>> os.environ['FOO'] = 'myvalue' 

307 >>> s.get_config_variable('foo') 

308 'myvalue' 

309 >>> s.set_config_variable('foo', 'othervalue') 

310 >>> s.get_config_variable('foo') 

311 'othervalue' 

312 

313 :type logical_name: str 

314 :param logical_name: The logical name of the session variable 

315 you want to set. These are the keys in ``SESSION_VARIABLES``. 

316 

317 :param value: The value to associate with the config variable. 

318 """ 

319 self._overrides[logical_name] = value 

320 

321 def clear_config_variable(self, logical_name): 

322 """Remove an override config variable from the session. 

323 

324 :type logical_name: str 

325 :param logical_name: The name of the parameter to clear the override 

326 value from. 

327 """ 

328 self._overrides.pop(logical_name, None)

331 """Set the provider for a config value. 

332 

333 This provides control over how a particular configuration value is 

334 loaded. This replaces the provider for ``logical_name`` with the new 

335 ``provider``. 

336 

337 :type logical_name: str

339 provider for. 

340 

341 :type provider: :class:`botocore.configprovider.BaseProvider` 

342 :param provider: The new provider that should be responsible for 

343 providing a value for the config named ``logical_name``. 

344 """ 

345 self._mapping[logical_name] = provider

359class ChainProvider(BaseProvider): 

360 """This provider wraps one or more other providers. 

361 

362 Each provider in the chain is called, the first one returning a non-None 

363 value is then returned. 

364 """ 

365 def __init__(self, providers=None, conversion_func=None): 

366 """Initalize a ChainProvider. 

367 

368 :type providers: list 

369 :param providers: The initial list of providers to check for values 

370 when invoked. 

371 

372 :type conversion_func: None or callable 

373 :param conversion_func: If this value is None then it has no affect on 

374 the return type. Otherwise, it is treated as a function that will 

375 transform provided value. 

376 """ 

377 if providers is None: 

378 providers = [] 

379 self._providers = providers 

380 self._conversion_func = conversion_func 

381 

382 def provide(self): 

383 """Provide the value from the first provider to return non-None. 

384 

385 Each provider in the chain has its provide method called. The first 

386 one in the chain to return a non-None value is the returned from the 

387 ChainProvider. When no non-None value is found, None is returned. 

388 """ 

389 for provider in self._providers: 

390 value = provider.provide() 

391 if value is not None: 

392 return self._convert_type(value) 

393 return None 

394 

395 def _convert_type(self, value): 

396 if self._conversion_func is not None: 

397 return self._conversion_func(value) 

398 return value 

399 

400 def __repr__(self): 

401 return '[%s]' % ', '.join([str(p) for p in self._providers])

432class ScopedConfigProvider(BaseProvider): 

433 def __init__(self, config_var_name, session): 

434 """Initialize ScopedConfigProvider. 

435 

436 :type config_var_name: str or tuple 

437 :param config_var_name: The name of the config variable to load from 

438 the configuration file. If the value is a tuple, it must only 

439 consist of two items, where the first item represents the section 

440 and the second item represents the config var name in the section. 

441 

442 :type session: :class:`botocore.session.Session` 

443 :param session: The botocore session to get the loaded configuration 

444 file variables from. 

445 """ 

446 self._config_var_name = config_var_name 

447 self._session = session 

448 

449 def provide(self): 

450 """Provide a value from a config file property.""" 

451 scoped_config = self._session.get_scoped_config() 

452 if isinstance(self._config_var_name, tuple): 

453 section_config = scoped_config.get(self._config_var_name[0]) 

454 if not isinstance(section_config, dict): 

455 return None 

456 return section_config.get(self._config_var_name[1]) 

457 return scoped_config.get(self._config_var_name) 

458

460 return 'ScopedConfigProvider(config_var_name=%s, session=%s)' % ( 

461 self._config_var_name,

490class SectionConfigProvider(BaseProvider): 

491 """Provides a dictionary from a section in the scoped config 

492 

493 This is useful for retrieving scoped config variables (i.e. s3) that have 

494 their own set of config variables and resolving logic. 

495 """ 

496 def __init__(self, section_name, session, override_providers=None): 

497 self._section_name = section_name 

498 self._session = session 

499 self._scoped_config_provider = ScopedConfigProvider( 

500 self._section_name, self._session) 

501 self._override_providers = override_providers 

502 if self._override_providers is None: 

503 self._override_providers = {} 

504 

505 def provide(self): 

506 section_config = self._scoped_config_provider.provide() 

507 if section_config and not isinstance(section_config, dict): 

508 logger.debug("The %s config key is not a dictionary type, " 

509 "ignoring its value of: %s", self._section_name, 

510 section_config) 

511 return None 

512 for section_config_var, provider in self._override_providers.items(): 

513 provider_val = provider.provide() 

514 if provider_val is not None: 

515 if section_config is None: 

516 section_config = {} 

517 section_config[section_config_var] = provider_val 

518 return section_config 

519 

520 def __repr__(self): 

521 return ( 

522 'SectionConfigProvider(section_name=%s, ' 

523 'session=%s, override_providers=%s)' % ( 

524 self._section_name, self._session, 

525 self._override_providers, 

526 ) 

527 )
