1"""Standard retry behavior. 

2 

3This contains the default standard retry behavior. 

4It provides consistent behavior with other AWS SDKs. 

5 

6The key base classes uses for retries: 

7 

8 * ``BaseRetryableChecker`` - Use to check a specific condition that 

9 indicates a retry should happen. This can include things like 

10 max attempts, HTTP status code checks, error code checks etc. 

11 * ``RetryBackoff`` - Use to determine how long we should backoff until 

12 we retry a request. This is the class that will implement delay such 

13 as exponential backoff. 

14 * ``RetryPolicy`` - Main class that determines if a retry should 

15 happen. It can combine data from a various BaseRetryableCheckers 

16 to make a final call as to whether or not a retry should happen. 

17 It then uses a ``BaseRetryBackoff`` to determine how long to delay. 

18 * ``RetryHandler`` - The bridge between botocore's event system 

19 used by endpoint.py to manage retries and the interfaces defined 

20 in this module. 

21 

22This allows us to define an API that has minimal coupling to the event 

23based API used by botocore. 

24 

25"""

64class RetryHandler(object): 

65 """Bridge between botocore's event system and this module. 

66 

67 This class is intended to be hooked to botocore's event system 

68 as an event handler. 

69 """ 

70 def __init__(self, retry_policy, retry_event_adapter, retry_quota): 

71 self._retry_policy = retry_policy 

72 self._retry_event_adapter = retry_event_adapter 

73 self._retry_quota = retry_quota 

74 

75 def needs_retry(self, **kwargs): 

76 """Connect as a handler to the needs-retry event.""" 

77 retry_delay = None 

78 context = self._retry_event_adapter.create_retry_context(**kwargs) 

79 if self._retry_policy.should_retry(context): 

80 # Before we can retry we need to ensure we have sufficient 

81 # capacity in our retry quota. 

82 if self._retry_quota.acquire_retry_quota(context): 

83 retry_delay = self._retry_policy.compute_retry_delay(context) 

84 logger.debug("Retry needed, retrying request after " 

85 "delay of: %s", retry_delay) 

86 else: 

87 logger.debug("Retry needed but retry quota reached, " 

88 "not retrying request.") 

89 else: 

90 logger.debug("Not retrying request.") 

91 self._retry_event_adapter.adapt_retry_response_from_context( 

92 context) 

93 return retry_delay

96class RetryEventAdapter(object): 

97 """Adapter to existing retry interface used in the endpoints layer. 

98 

99 This existing interface for determining if a retry needs to happen 

100 is event based and used in ``botocore.endpoint``. The interface has 

101 grown organically over the years and could use some cleanup. This 

102 adapter converts that interface into the interface used by the 

103 new retry strategies. 

104 

105 """ 

106 def create_retry_context(self, **kwargs): 

107 """Create context based on needs-retry kwargs.""" 

108 response = kwargs['response'] 

109 if response is None: 

110 # If response is None it means that an exception was raised 

111 # because we never received a response from the service. This 

112 # could be something like a ConnectionError we get from our 

113 # http layer. 

114 http_response = None 

115 parsed_response = None 

116 else: 

117 http_response, parsed_response = response 

118 # This provides isolation between the kwargs emitted in the 

119 # needs-retry event, and what this module uses to check for 

120 # retries. 

121 context = RetryContext( 

122 attempt_number=kwargs['attempts'], 

123 operation_model=kwargs['operation'], 

124 http_response=http_response, 

125 parsed_response=parsed_response, 

126 caught_exception=kwargs['caught_exception'], 

127 request_context=kwargs['request_dict']['context'], 

128 ) 

129 return context

132 """Modify response back to user back from context.""" 

133 # This will mutate attributes that are returned back to the end

138 context.parsed_response.setdefault(

145class RetryContext(object): 

146 """Normalize a response that we use to check if a retry should occur. 

147 

148 This class smoothes over the different types of responses we may get 

149 from a service including: 

150 

151 * A modeled error response from the service that contains a service 

152 code and error message. 

153 * A raw HTTP response that doesn't contain service protocol specific 

154 error keys. 

155 * An exception received while attempting to retrieve a response. 

156 This could be a ConnectionError we receive from our HTTP layer which 

157 could represent that we weren't able to receive a response from 

158 the service. 

159 

160 This class guarantees that at least one of the above attributes will be 

161 non None. 

162 

163 This class is meant to provide a read-only view into the properties 

164 associated with a possible retryable response. None of the properties 

165 are meant to be modified directly. 

166 

167 """ 

168 def __init__(self, attempt_number, operation_model=None, 

169 parsed_response=None, http_response=None, 

170 caught_exception=None, request_context=None): 

171 # 1-based attempt number. 

172 self.attempt_number = attempt_number 

173 self.operation_model = operation_model 

174 # This is the parsed response dictionary we get from parsing 

175 # the HTTP response from the service. 

176 self.parsed_response = parsed_response 

177 # This is an instance of botocore.awsrequest.AWSResponse. 

178 self.http_response = http_response 

179 # This is a subclass of Exception that will be non None if 

180 # an exception was raised when retrying to retrieve a response. 

181 self.caught_exception = caught_exception 

182 # This is the request context dictionary that's added to the 

183 # request dict. This is used to story any additional state 

184 # about the request. We use this for storing retry quota 

185 # capacity. 

186 if request_context is None: 

187 request_context = {} 

188 self.request_context = request_context 

189 self._retry_metadata = {} 

190 

191 # These are misc helper methods to avoid duplication in the various 

192 # checkers. 

193 def get_error_code(self): 

194 """Check if there was a parsed response with an error code. 

195 

196 If we could not find any error codes, ``None`` is returned. 

197 

198 """ 

199 if self.parsed_response is None: 

200 return 

201 return self.parsed_response.get('Error', {}).get('Code')

230class ExponentialBackoff(BaseRetryBackoff): 

231 

232 _BASE = 2 

233 _MAX_BACKOFF = 20 

234 

235 def __init__(self, max_backoff=20, random=random.random): 

236 self._base = self._BASE 

237 self._max_backoff = max_backoff 

238 self._random = random 

239 

240 def delay_amount(self, context): 

241 """Calculates delay based on exponential backoff. 

242 

243 This class implements truncated binary exponential backoff 

244 with jitter:: 

245 

246 t_i = min(rand(0, 1) * 2 ** attempt, MAX_BACKOFF) 

247 

248 where ``i`` is the request attempt (0 based). 

249 

250 """ 

251 # The context.attempt_number is a 1-based value, but we have 

252 # to calculate the delay based on i based a 0-based value. We 

253 # want the first delay to just be ``rand(0, 1)``. 

254 return min( 

255 self._random() * (self._base ** (context.attempt_number - 1)), 

256 self._max_backoff 

257 )

354class ModeledRetryErrorDetector(object): 

355 """Checks whether or not an error is a modeled retryable error.""" 

356 # There are return values from the detect_error_type() method. 

357 TRANSIENT_ERROR = 'TRANSIENT_ERROR' 

358 THROTTLING_ERROR = 'THROTTLING_ERROR' 

359 # This class is lower level than ModeledRetryableChecker, which 

360 # implements BaseRetryableChecker. This object allows you to distinguish 

361 # between the various types of retryable errors. 

362 

363 def detect_error_type(self, context): 

364 """Detect the error type associated with an error code and model. 

365 

366 This will either return: 

367 

368 * ``self.TRANSIENT_ERROR`` - If the error is a transient error 

369 * ``self.THROTTLING_ERROR`` - If the error is a throttling error 

370 * ``None`` - If the error is neither type of error. 

371 

372 """ 

373 error_code = context.get_error_code() 

374 op_model = context.operation_model 

375 if op_model is None or not op_model.error_shapes: 

376 return 

377 for shape in op_model.error_shapes: 

378 if shape.metadata.get('retryable') is not None: 

379 # Check if this error code matches the shape. This can 

380 # be either by name or by a modeled error code. 

381 error_code_to_check = ( 

382 shape.metadata.get('error', {}).get('code') 

383 or shape.name 

384 ) 

385 if error_code == error_code_to_check: 

386 if shape.metadata['retryable'].get('throttling'): 

387 return self.THROTTLING_ERROR 

388 return self.TRANSIENT_ERROR

442class RetryQuotaChecker(object): 

443 _RETRY_COST = 5 

444 _NO_RETRY_INCREMENT = 1 

445 _TIMEOUT_RETRY_REQUEST = 10 

446 _TIMEOUT_EXCEPTIONS = (ConnectTimeoutError, ReadTimeoutError) 

447 

448 # Implementation note: We're not making this a BaseRetryableChecker 

449 # because this isn't just a check if we can retry. This also changes 

450 # state so we have to careful when/how we call this. Making it 

451 # a BaseRetryableChecker implies you can call .is_retryable(context) 

452 # as many times as you want and not affect anything. 

453 

454 def __init__(self, quota): 

455 self._quota = quota 

456 # This tracks the last amount 

457 self._last_amount_acquired = None 

458 

459 def acquire_retry_quota(self, context): 

460 if self._is_timeout_error(context): 

461 capacity_amount = self._TIMEOUT_RETRY_REQUEST 

462 else: 

463 capacity_amount = self._RETRY_COST 

464 success = self._quota.acquire(capacity_amount) 

465 if success: 

466 # We add the capacity amount to the request context so we know 

467 # how much to release later. The capacity amount can vary based 

468 # on the error. 

469 context.request_context['retry_quota_capacity'] = capacity_amount 

470 return True 

471 context.add_retry_metadata(RetryQuotaReached=True) 

472 return False 

473 

474 def _is_timeout_error(self, context): 

475 return isinstance(context.caught_exception, self._TIMEOUT_EXCEPTIONS) 

476 

477 # This is intended to be hooked up to ``after-call``. 

478 def release_retry_quota(self, context, http_response, **kwargs): 

479 # There's three possible options. 

480 # 1. The HTTP response did not have a 2xx response. In that case we 

481 # give no quota back. 

482 # 2. The HTTP request was successful and was never retried. In 

483 # that case we give _NO_RETRY_INCREMENT back. 

484 # 3. The API call had retries, and we eventually receive an HTTP 

485 # response with a 2xx status code. In that case we give back 

486 # whatever quota was associated with the last acquisition. 

487 if http_response is None: 

488 return 

489 status_code = http_response.status_code 

490 if 200 <= status_code < 300: 

491 if 'retry_quota_capacity' not in context: 

492 self._quota.release(self._NO_RETRY_INCREMENT) 

493 else: 

494 capacity_amount = context['retry_quota_capacity'] 

495 self._quota.release(capacity_amount)
